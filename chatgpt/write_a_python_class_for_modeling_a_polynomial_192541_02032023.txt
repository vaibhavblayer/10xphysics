Q. Write a python class for modeling a polynomial.

class Polynomial:
    def __init__(self, coefficients):
        self.coefficients = coefficients
 
    def evaluate(self, x):
        result = 0
        for i, c in enumerate(self.coefficients):
            result += c * x**i
        return result
 
    def __str__(self):
        terms = [str(self.coefficients[0])]
        for i, c in enumerate(self.coefficients[1:], 1):
            if c == 0:
                continue
            if c > 0:
                sign = '+'
            else:
                sign = '-'
            terms.append(f'{sign} {abs(c)}x^{i}')
        return ' '.join(terms)
