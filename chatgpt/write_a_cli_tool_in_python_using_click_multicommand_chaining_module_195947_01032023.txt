Q. Write a cli tool in python using click multicommand chaining module.

import click

@click.group()
def cli():
    pass

@cli.command()
@click.option('--name', prompt='What is your name?',
              help='The person to greet.')
def hello(name):
    click.echo('Hello %s!' % name)

@cli.command()
@click.option('--name', prompt='What is your name?',
              help='The person to greet.')
@click.option('--message', prompt='What message do you want to send?',
              help='The message to send.')
def send(name, message):
    click.echo('Sending message to %s: %s' % (name, message))

if __name__ == '__main__':
    cli()
