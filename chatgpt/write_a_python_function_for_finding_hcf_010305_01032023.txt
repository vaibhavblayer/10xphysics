Q. Write a python function for finding HCF.

def hcf(x, y):
  
  # choose the smaller number
  if x > y:
    smaller = y
  else:
    smaller = x
  
  # loop through the range of the smaller number
  for i in range(1, smaller+1):
    if((x % i == 0) and (y % i == 0)):
      hcf = i
      
  return hcf

# test the function
num1 = 54 
num2 = 24

print("The HCF of", num1,"and", num2,"is", hcf(num1, num2))
