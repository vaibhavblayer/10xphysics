Q. Write python code for animating a point along a circle using manim library.

# In[1]:


from manimlib.imports import *


# In[2]:


class Circle(Scene):
    def construct(self):
        circle = Circle()
        self.play(ShowCreation(circle))
        self.wait(2)


# In[3]:


class SquareToCircle(Scene):
    def construct(self):
        circle = Circle()
        square = Square()
        square.flip(RIGHT)
        square.rotate(-3 * TAU / 8)
        circle.set_fill(PINK, opacity=0.5)

        self.play(ShowCreation(square))
        self.play(Transform(square, circle))
        self.play(FadeOut(square))


# In[4]:


class WarpSquare(Scene):
    def construct(self):
        square = Square()
        self.play(ApplyPointwiseFunction(
            lambda point: complex_to_R3(np.exp(R3_to_complex(point))),
            square
        ))
        self.wait(2)


# In[5]:


class WriteStuff(Scene):
    def construct(self):
        example_text = TextMobject("This is a some text")
        example_tex = TexMobject(
            "\\sum_{k=1}^\\infty {1 \\over k^2} = {\\pi^2 \\over 6}"
        )
        group = VGroup(example_text, example_tex)
        group.arrange(DOWN)
        self.play(Write(example_text))
        self.play(Write(example_tex))
        self.wait(2)


# In[6]:


class UpdatersExample(Scene):
    def construct(self):
        decimal = DecimalNumber(
            0,
            show_ellipsis=True,
            num_decimal_places=3,
            include_sign=True,
        )
        square = Square().to_edge(UP)

        decimal.add_updater(lambda d: d.next_to(square, RIGHT))
        decimal.add_updater(lambda d: d.set_value(square.get_center()[1]))
        self.add(square, decimal)
        self.play(
            square.to_edge, DOWN,
            rate_func=there_and_back,
            run_time=5,
        )
        self.wait()

class Shapes(Scene):
    def construct(self):
        circle = Circle()
        square = Square()
        square.flip(RIGHT)
        square.rotate(-3 * TAU / 8)
        circle.set_fill(PINK, opacity=0.5)

        circle.shift(2 * DOWN)
        square.shift(2 * UP)

        self.play(ShowCreation(VGroup(circle, square)))
        self.wait(2)

class Circles(Scene):
    def construct(self):
        circle1 = Circle()
        circle2 = Circle()
        circle2.shift(2 * UP)
        circle3 = Circle()
        circle3.shift(2 * DOWN)
        circle4 = Circle()
        circle4.shift(2 * LEFT)
        circle5 = Circle()
        circle5.shift(2 * RIGHT)

        self.play(ShowCreation(VGroup(circle1, circle2, circle3, circle4, circle5)))
        self.wait(2)
