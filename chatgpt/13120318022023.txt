Q. Write python code for animating a helix using manim library.

# In[1]:


from manimlib.imports import *


# In[2]:


class Helix(ThreeDScene):
    def construct(self):
        axes = ThreeDAxes()
        helix = ParametricFunction(
            lambda u: np.array([
                np.cos(u),
                np.sin(u),
                u
            ]), color=RED, t_min=0, t_max=TAU,
        )
        self.set_camera_orientation(phi=75 * DEGREES)
        self.play(ShowCreation(axes))
        self.play(ShowCreation(helix))
        self.wait(2)


# In[3]:


scene = Helix()
scene.write_gif()
