\begin{minted}[tabsize=4, breaklines, mathescape, escapeinside=\[\]]{python}


def derivative(f, x, h):
	
	# $f'(x)=\dfrac{f(x+h)-f(x)}{h}$
	
    return (f(x + h) - f(x)) / h

#Example
def f(x):
    return x**2

derivative(f, 2, 0.01) 
# Output: 4.01
\end{minted}
