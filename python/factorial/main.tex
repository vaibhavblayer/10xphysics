\begin{minted}[tabsize=4, breaklines, mathescape]{python}

def factorial(n):
	
	# $n! = n\cdot(n-1)!$
	# $0! = 1$

    if n == 0:
        return 1
    else:
        return n * factorial(n-1)
      
factorial(5) 
 
# Output:
# 120
\end{minted}
