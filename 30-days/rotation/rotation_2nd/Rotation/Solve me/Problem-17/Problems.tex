\begin{question}
A uniform ring of mass $m$ and radius $R$ has  a particle of mass $2m$ rigidly attached to it on the inside surface. The system rolls on a horizontal surface without slipping. In the position shown at a certain instant, its centre $C$ has a velocity $v_0$. Find the kinetic energy of the system.
\end{question}

{\physics}

\begin{center}
\begin{tikzpicture}[>=Stealth,very thick,every node/.style={scale=0.75}]
\draw[mirror->] (-3,0)--(3,0);
\draw (0,1.25) edge[->] node[at end,above,red]{$v_0$} (2.25,1.25) circle[radius=1.25];
\fill [red] (0,1.25) node[left]{$C$} circle(2pt);
\fill [black] (0,0.15) node[above=2 mm,red]{$2m$} circle(4pt);
\node at (1,2)[right,red]{$m$};
\node at (0,0)[below=2 mm,red]{$P$};
\end{tikzpicture}
\end{center}

\pagebreak


\pagestyle{empty}

\begin{center}
{\solution}
\end{center}

\begin{calculations}
As the system rotates point $P$ becomes rest for a moment, therefore only ring will have some velocity hence the kinetic energy. 

\begin{align*}
K.E_\textit{total} &= K.E_\textit{ring} + K.E_\textit{particle} \\[2 mm]
						 &= K.E_\textit{ring} + 0 \\[2 mm]
						 &= \dfrac{1}{2}I\omega^2 + \dfrac{1}{2}mv_0^2 \\[2 mm]
						 &= \dfrac{1}{2} mR^2 \left( \dfrac{v_0^2}{R^2} \right) + \dfrac{1}{2}mv_0^2 \\[2 mm]
						 &= mv_0^2  \ans
\end{align*}
\end{calculations}
