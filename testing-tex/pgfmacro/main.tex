\def\pgfpointadd#1#2{%
 \pgf@process{#1}% 
 \pgf@xa=\pgf@x% 
 \pgf@ya=\pgf@y% 
 \pgf@process{#2}% 
 \advance\pgf@x by\pgf@xa% 
 \advance\pgf@y by\pgf@ya}
 
 
 
\makeatletter
\newcommand\xcoord[2][center]{{%
    \pgfpointxy{1}{1}%
    \@tempdima=\pgf@x
    \pgfpointanchor{#2}{#1}%
    \@tempdimb=\pgf@x
    \pgfmathparse{\@tempdimb/\@tempdima}%
    \num{\pgfmathresult}%
}}
\newcommand\ycoord[2][center]{{%
    \pgfpointxy{1}{1}%

    \pgfpointanchor{#2}{#1}%
    \@tempdimb=\pgf@y
    \pgfmathparse{\@tempdimb/\@tempdima}%
    \num{\pgfmathresult}%
}}
