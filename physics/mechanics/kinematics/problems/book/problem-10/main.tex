	\tztangentat[->]"tangent"{curve}{-\r*sin{60}}[-\r:-\r+1]{$r\omega$}
	\tzvXpointat*{curve}{-\r*sin{60}}(A){$A$}[l]
	
	\tzline+[->](A)(\r + 0.35, 0){$v_0$}[r]
	\tzanglemark(A)(O)($(O)+(0, -\r)$){$\theta$}
	\tzline[dashed](A)(O)
	\tzline[dashed](O)($(O)+(0, -\r)$)
	\tzanglemark($(A)+(1, 0)$)(A)($(A)+(0.2, 0.35)$){$180^\circ - \theta$}[scale=0.75, r](12pt)
	
\end{tikzpicture}
\end{center}

\begin{align*}
\vec{v}_A &= \vec{v}_t + \vec{v}_O \\
v_A &= \sqrt{v^2_t + v^2_0 + 2 \cdot v_t \cdot v_0 \cos\left(180^\circ -\theta\right)} \intertext{For pure rolling motion $v_0 = r\omega$}\\
v_A &= \sqrt{\left(r\omega\right)^2 + v^2_0 - 2 \cdot r\omega \cdot v_0 \cos\theta}\\
v_A &= \sqrt{v_0^2 + v^2_0 - 2 \cdot v_0 \cdot v_0 \cos\theta} \\
\Aboxed{v_A &= 2v_0\sin\left(\dfrac{\theta}{2}\right)}\\[-15mm]
\end{align*}
\pagebreak

\begin{align*}
d &= \int_0^{T} |v_A| \d{t} \intertext{angular displacement ~ $\theta = \omega t$, ~$v_A >0$ as ~$\sin(\theta/2)$ ~ remains +ve}\\
  &= \int_0^{T} 2v_0\sin\left(\dfrac{\theta}{2}\right)\d{t} \\
  &= \int_0^{T} 2v_0\sin\left(\dfrac{\omega t}{2}\right)\d{t} \\
  &= 2v_0 \left[ \dfrac{-\cos\left(\dfrac{\omega t}{2}\right)}{\dfrac{\omega}{2}} \right]_0^T \intertext{time period ~ $T=2\pi/\omega$}\\
  &=\dfrac{4v_0}{\omega} \left[ -\cos\left(\dfrac{\omega}{2}\cdot \dfrac{2\pi}{\omega}\right) + 1 \right]\\
  \Aboxed{d &= 8r} \ans
\end{align*}
