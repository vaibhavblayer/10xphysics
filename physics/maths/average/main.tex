	\def\Fn{0.25*\x*\x - \x + 2 - sin(deg(1.5*\x))}


	\tzfn\Fn[0:5.5]
	%\tzfn[dashed]\Fnn[0:6.3]
	
	\tzvXpoint{Fn}(3,1)(A)
	\tzvXpoint{Fn}(3.2,1)(B)
	\tzvXpoint{Fn}(5.4,1)(C)
	
	\tzvXpoint{Fn}(2,1)(x)
	\tzvXpoint{Fn}(4.5,1)(X)
	%\tzdots*(A)(B);
	\tzproj*[<-](x){$x_1$}
	\tzproj*[<-](X){$x_2$}
	
	\tzfnarea*[pattern=dots]{\Fn}[2:4.5]
	
	\draw[pattern, pattern=north east lines](3, 0)--(3.2, 0)--(B)--(A)--cycle;
	
	\tznode(3.1, 0){$d\!x$}[b]
	
	\tzlines+[<-](C)(0.5, -0.5)(1, 0){$f(x)$}[r];
\end{tikzpicture}
\end{center}
\vspace*{\fill}
\begin{align*}
< f(x) > = \dfrac{\int_{x_1}^{x_2} f(x) d\!x}{\int_{x_1}^{x_2} d\!x}
\end{align*}

	\tzvXpoint{Fn}(3,1)(A)

	\tzproj*[<-](X){$t_2$}

< \vec{v} > = \dfrac{\int_{t_1}^{t_2} \vec{v} d\!t}{\int_{t_1}^{t_2} d\!t}
\end{align*}
