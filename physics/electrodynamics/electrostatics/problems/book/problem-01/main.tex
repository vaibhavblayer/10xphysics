\def\question{
Two identical balls each having a density $\uprho$ are suspended from a common point by two insulating strings of equal length. Both the balls have equal mass and charge. In equilibrium each string makes an angle $\uptheta$ with vertical. Now, both the balls are immersed in a liquid. As a result the angle $\uptheta$ does not change. The density of the liquid is $\upsigma$. Find the dielectric constant of the liquid.
}

\begin{center}
\begin{tikzpicture}

	\tzline[-->--=0.4](Q1)(O){$T$}[pos=0.35, a]
	\tzline(O)(Q2)
	\tzanglemark(Q1)(O)(O'){$\theta$}
	\tzanglemark(O)(Q1)($(Q1)+(0, 1)$){$\theta$}
	\tzline[dashed](Q1)($(Q1)+(0, 2)$)
	\tzline[dashed](O)(O')
	\tzline+[->](Q1)(0, -1){$W$}[b]
	\tzline+[->](Q1)(-1, 0){$F_e$}[l]
\end{tikzpicture}
\end{center}

\begin{align*}
W &= \textit{weight}\\
F_e &= \textit{electrostatic force}\\
T &= \textit{Tension}
\end{align*}

\begin{align}
\intertext{Apply Lemi's theorem in vacuum}
\dfrac{W}{\sin\left(90^\circ + \theta\right)} = 
\dfrac{F_e}{\sin\left(180^\circ-\theta\right)} =
\dfrac{T}{\sin\left(90^\circ\right)}
\end{align}

\end{tikzpicture}

\begin{align*}
W &= \textit{weight}\\
F_b &= \textit{buoyant force}\\
F'_e &= \textit{electrostatic force in the liquid}\\
T' &= \textit{Tension}
\end{align*}

\begin{align}
\intertext{Apply Lemi's theorem in liquid}
\dfrac{W-F_b}{\sin\left(90^\circ + \theta\right)} = 
\dfrac{F'_e}{\sin\left(180^\circ-\theta\right)} =
\dfrac{T'}{\sin\left(90^\circ\right)}
\end{align}
\pagebreak

\addtolength{\jot}{2ex}
\begin{align*}
\intertext{From equation (1) and (2) $\rightarrow$ (2)/(1)}
\dfrac{W-F_b}{W} &= \dfrac{F'_e}{F_e}\\
\dfrac{V\uprho g - V\upsigma g}{V\uprho g} &= \dfrac{F_e}{K\cdot F_e}\\
\dfrac{\uprho-\upsigma}{\uprho} &= \dfrac{1}{K}\\
\Aboxed{K &= \dfrac{\uprho}{\uprho-\upsigma}}\ans
\end{align*}
