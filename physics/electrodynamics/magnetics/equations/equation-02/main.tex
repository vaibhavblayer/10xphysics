\begin{align*}
W & =\int \vec{F} \cdot \d{\vec{r}}\\
 & = \int q_0 \left( \vec{v} \times \vec{B} \right) \cdot \vec{v}\d{t}\\
 & = \int q_0 \left[\vec{v} \vec{B} \vec{v} \right] \d{t}\\
W & = 0
\end{align*}
