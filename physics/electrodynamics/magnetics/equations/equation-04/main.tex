\begin{align*}
T & = \dfrac{2\pi}{\omega}\\
  & = \dfrac{2\pi}{\left(\dfrac{v}{r}\right)} &&\omega = \dfrac{v}{r}\\
  & =\dfrac{2\pi r}{v} &&r=\dfrac{mv}{qB}\\
T & = \dfrac{2\pi m}{qB}
\end{align*}
