\begin{align*}
\vec{F}_{\textit{magnetic}} & =\vec{F}_{\textit{centripetal}}\\
qvB & = \dfrac{mv^2}{r} \\
r & = \dfrac{mv^2}{qvB}\\
r & = \dfrac{mv}{qB}
\end{align*}
