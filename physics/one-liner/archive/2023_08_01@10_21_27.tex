\begin{align*}
Q &.~\lim_{x\to 1}\dfrac{x + x^2 + \cdots + x^n-n}{x-1}\\
	&\Rightarrow \lim_{x\to 1}\dfrac{x + x^2 + \cdots + x^n-n}{x-1} &\left[\dfrac{0}{0} \text{ form}\right]\\
	&= \lim_{x\to 1}\dfrac{1 + 2x + \cdots + nx^{n-1}}{1} &\left[\dfrac{f'(x)}{g'(x)}\right]\\
	&=1+2+\cdots+n\\
	&= \dfrac{n(n+1)}{2}
\end{align*}
