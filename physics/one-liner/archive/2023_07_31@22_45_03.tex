\begin{align*}
Q &.~\lim_{x\to 2}\dfrac{x^6-24x-16}{x^3+2x-12}\\
	&\Rightarrow \lim_{x\to 2}\dfrac{x^6-24x-16}{x^3+2x-12} &\left[\dfrac{0}{0} \text{ form}\right]\\
	&= \lim_{x\to 2}\dfrac{6x^5-24}{3x^2+2} &\left[\dfrac{f'(x)}{g'(x)}\right]\\
	&=\dfrac{6(2)^5-24}{3(2)^2+2}\\
	&= \dfrac{168}{14}\\
	&=12
\end{align*}
