\begin{align*}
-\dfrac{\d{N}}{\d{t}} &\propto N \\
-\dfrac{\d{N}}{\d{t}} &= \lambda N\\
\int_{N_0}^{N} \dfrac{\d{N}}{N} &= -\lambda \int_0^t \d{t}\\
\ln\left( \dfrac{N}{N_0} \right) &= -\lambda t \\
\Aboxed{N &= N_0e^{-\lambda t}}
\end{align*}
