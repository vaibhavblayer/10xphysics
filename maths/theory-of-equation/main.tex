\begin{enumerate}
\item[Q.] Find $n$ for which 
\[
3^9+3^{12}+3^{15}+3^n
\]
is a perfect cube.

\begin{enumerate}
\item[$\Rightarrow$] 
\addtolength{\jot}{2ex}
\begin{align*}
& 3^9+3^{12}+3^{15}+3^n\\
= & 3^9\left(1+3^3+3^6+3^{n-9}\right)\\
= & \left(3^3\right)^3\{ 1 + 3\cdot3^2 + 3\cdot\left(3^2\right)^2 + \left(3^2\right)^3 + 3^{n-9}  -3\cdot\left(3^2\right)^2\}\\
= & \left(3^3\right)^3\{\left(1+3^2\right)^3 +3^{n-9}  -3^5\}\\
\intertext{$3^{n-9}  -3^5$ has to be zero}
3^{n-9}  -3^5 &= 0\\
3^{n-9} &= 3^5 \\
\Aboxed{n &= 14}
\end{align*}
\end{enumerate}
\end{enumerate}
